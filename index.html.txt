<script>
const clans = {
  brujah: {
    name: "Brujah",
    desc: "Бунтари и революционеры.",
    link: "brujah.html"
  },
  toreador: {
    name: "Toreador",
    desc: "Любители искусства и красоты.",
    link: "toreador.html"
  },
  ventrue: {
    name: "Ventrue",
    desc: "Элита и правители.",
    link: "ventrue.html"
  }
};

function showClan(id) {
  const clan = clans[id];

  document.getElementById("clanName").innerText = clan.name;
  document.getElementById("clanDesc").innerText = clan.desc;

  const btn = document.getElementById("moreBtn");
  btn.style.display = "inline-block";
  btn.onclick = () => {
    window.location.href = clan.link;
  };
}
</script>
